\makeglossaries

\newglossaryentry{wallet}
{
    name=Wallet,
    description={A traditional wallet is a program that stores a secret key and signs transactions to interact with a blockchain. Hardware wallets are a variation that keeps the secret on a dedicated device}
}
\newglossaryentry{beneficiary}
{
    name=Beneficiary,
    text={beneficiary},
    description={In this context a beneficiary is any other person, represented by a traditional wallet, who may receive funds once the proof-of-life window (and any personal delay) has lapsed}
}
\newglossaryentry{owner}
{
    name=Owner,
    text={owner},
    description={In this context an owner is a person with admin or sufficient agreeing multi-signature users with power over the wallet. By default the wallet has a single admin owner with broad access to the funds; this access can be restricted, for example by requiring a multi-signature quorum across several owners}
}
\newglossaryentry{smartcontracts}
{
    name=Smart Contract,
    description={Software whose execution is validated by a decentralized network, so its results can be trusted; deployed on a blockchain to offer on-chain functionality}
}
\newglossaryentry{smartvalidator}
{
    name=Validator,
    description={The EUTXO counterpart of a smart contract: a script attached to an output that runs when the output is spent and accepts or rejects the whole transaction}
}
\newglossaryentry{dApp}
{
    name=dApp,
    description={A decentralized application: software, typically a web app, through which users interact with a blockchain and its smart contracts}
}
\newglossaryentry{vesting}
{
    name=Vesting,
    description={Locking funds until a chosen date, before which the beneficiary (or owner) cannot access them}
}
\newglossaryentry{state-machine}
{
    name=State Machine,
    description={A computer-science model that describes a system as a set of states and the transitions between them; used here to model the wallet's reachable states}
}
\newglossaryentry{proof-of-life}
{
    name=Proof of Life,
    text={proof of life},
    description={Proof of Life is the dead-man-switch heartbeat: a renewal of the wallet's unlock-time deadline. It is satisfied by any operator action that renews the window, or by a dedicated liveness-keeper renewal transaction through the dApp. If it is not provided before the deadline, beneficiaries may unlock}
}
\newglossaryentry{stt}
{
    name=State-Thread Token (STT),
    text={STT},
    first={state-thread token (STT)},
    description={A single native token, unique per wallet, whose continuing UTXO carries the wallet's authoritative configuration (the State) as its datum. Exactly one exists per wallet and it must be forwarded on every transition, so it is always the single source of truth for that wallet's rules}
}
\newglossaryentry{streaming-payment}
{
    name=Streaming Payment,
    description={A recurring payout that accrues linearly between a start and end date to a fixed payee. The accrued-but-unpaid amount may be settled by anyone (a permission-less crank), since the contract fixes the destination and caps the amount at what has accrued, but also guarantees accrued funds always remain accessible to the payee and cannot be diverted by anyone}
}
